## 冒泡排序
#### 
````javascript
function checkArray(arr) {
    if(!arr || arr.length <= 2) {
        return arr
    }
}
````

````javascript
function bulleSort(arr){
    checkArray(arr);
    for(let j = 0; j < arr.length; j++) {
        for(let i = 0; i < arr.length -1; i++) {
            if(arr[i] > arr[i+1]) {
                let tmp = arr[i];
                arr[i] = arr[i+1];
                arr[i+1] = tmp;
            }
        }
    }
    return arr;
}
````

## 快速排序
####
````javascript
function quickSort(arr) {
    checkArray(arr);
    const p = arr[arr.length - 1];
    const left = arr.filter((item, index) => item <= p && index !== arr.length -1;
    const right = arr.filter((item, index) => item > p);
    return [...quickSort(left), p, ...quickSort(right)]
}
````

## 归并排序
####
````javascript
function merge(left, right) {
    let  i = 0;
    let j = 0;
    let res = [];
    while(i <= left.length && j <= right.length) {
        if (left[i] <= right[j]) {
            res.push(left[i]);
            i++
        } else {
            res.push(right[j]);
            j++
        }
    }
    if(i < left.length) {
        res.concat(left.slice(i))
    }
    if(j < right.length) {
        res.concat(right.slice(j))
    }
    return res
}

function mergeSort(arr) {
    checkArray(arr);
    let tmp = Math.floor(arr.length / 2);
    const left = mergeSort(arr.slice(0, tmp));
    const right = mergeSort(arr.slice(tmp));
    return merge(left, right)
}
````

## 插入排序
#### 将目标元素插入到有序数组中的合适位置,从有序数组的最后一个元素开始比较，然后后移操作
````javascript
function insertSort(arr) {
    checkArray(arr);
    for (let i = 1; i < arr.length; i++) {
        let j = i;
        let target = arr[j];
        while(j > 0 && arr[j - 1] > target) {
            arr[j] = arr[j - 1]
            j--
        }
        arr[j] = target
    }  
}
````

## 选择排序
#### 选择出一个最小值，放到有序数组的最后
````javascript
function selectSort(arr) {
    checkArray(arr);
    for(let i = 0; i < arr.length; i++) {
        let minIndex = i;
        for(let j = i; j < arr.length; j++) {
            if (arr[minIndex] > arr[j]) {
                minIndex = j;
            }
        }
        if(minIndex !== i) {
           let tmp = arr[minIndex];
           arr[minIndex] = arr[i];
           arr[i] = tmp;
        }
    }
    return arr
}
````
