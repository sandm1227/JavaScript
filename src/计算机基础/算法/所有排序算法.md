## 冒泡排序
#### 
````javascript
function checkArray(arr) {
    if(!arr || arr.length <= 2) {
        return arr
    }
}
````

````javascript
function bulleSort(arr){
    checkArray(arr)
    for(let j = 0; j < arr.length; j++) {
        for(let i = 0; i < arr.length -1; i++) {
            if(arr[i] > arr[i+1]) {
                let tmp = arr[i];
                arr[i] = arr[i+1];
                arr[i+1] = tmp;
            }
        }
    }
    return arr;
}
````
