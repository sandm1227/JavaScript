## 树的遍历
#### 树的深度优先遍历
##### 递归实现
````javascript
function bltree(root) {
    console.log(root);
    bltree(root.left);
    bltree(root.right);
}
````
#### 非递归 - 先序遍历（根左右）
````javascript
function bltree(root) {
    if(root) {
        let stack = [];
        stack.push(root);

        while(stack.length > 0) {
            root = stack.pop();
            console.log(root.value);

            if(root.right) {
                stack.push(root.right)
            }
            if(root.left) {
                stack.push(root.left)
            }
        }
    }
}
````

#### 树的广度优先遍历
````javascript
function bltree(root) {
    if(root) {
        let stack = [];
        stack.push(root);

        while(stack.length > 0) {
            root = stack.shift();
            console.log(root.value);

            if(root.left) {
                stack.push(root.left)
            }
            if(root.right) {
                stack.push(root.right)
            }
        }
    }
}
````
